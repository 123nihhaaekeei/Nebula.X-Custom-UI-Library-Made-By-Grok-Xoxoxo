-- Utility Functions
do
    function NebulaX:Connect(Signal, Callback)
        local Connection = Signal:Connect(Callback)
        self.Connections[#self.Connections + 1] = Connection
        return Connection
    end

    function NebulaX:Disconnect(Connection)
        Connection:Disconnect()
    end

    function NebulaX:Round(Number, Factor)
        return math.floor(Number / Factor) * Factor
    end

    function NebulaX:NextFlag()
        self.UnnamedFlagCount = self.UnnamedFlagCount + 1
        return string.format("Flag_%d", self.UnnamedFlagCount)
    end

    function NebulaX:RGBA(r, g, b, alpha)
        local Color = Color3.fromRGB(r, g, b)
        local Meta = table.clone(getrawmetatable(Color))
        setreadonly(Meta, false)
        local OldIndex = Meta.__index

        Meta.__index = newcclosure(function(self, key)
            if key:lower() == "transparency" then
                return alpha
            end
            return OldIndex(self, key)
        end)

        setrawmetatable(Color, Meta)
        return Color
    end

    function NebulaX:GetConfig()
        local Config = ""
        for Index, Value in pairs(self.Flags) do
            if not (Index == "ConfigConfig_List" or Index == "ConfigConfig_Load" or Index == "ConfigConfig_Save") then
                local Final = ""
                if typeof(Value) == "Color3" then
                    local h, s, v = Value:ToHSV()
                    Final = string.format("rgb(%s,%s,%s,%s)", h, s, v, 1)
                elseif typeof(Value) == "table" and Value.Color and Value.Transparency then
                    local h, s, v = Value.Color:ToHSV()
                    Final = string.format("rgb(%s,%s,%s,%s)", h, s, v, Value.Transparency)
                elseif typeof(Value) == "table" and Value.Mode then
                    Final = string.format("key(%s,%s,%s)", Value.current[1] or "nil", Value.current[2] or "nil", Value.Mode)
                elseif Value ~= nil then
                    if typeof(Value) == "boolean" then
                        Final = string.format("bool(%s)", tostring(Value))
                    elseif typeof(Value) == "table" then
                        local New = "table("
                        for _, v in pairs(Value) do
                            New = New .. v .. ","
                        end
                        New = New:sub(1, -2) .. ")"
                        Final = New
                    elseif typeof(Value) == "string" then
                        Final = string.format("string(%s)", Value)
                    elseif typeof(Value) == "number" then
                        Final = string.format("number(%s)", Value)
                    end
                end
                Config = Config .. Index .. ": " .. tostring(Final) .. "\n"
            end
        end
        return Config
    end

    function NebulaX:LoadConfig(Config)
        local Lines = string.split(Config, "\n")
        local Parsed = {}
        for _, Line in pairs(Lines) do
            local Parts = string.split(Line, ":")
            if Parts[1] ~= "ConfigConfig_List" and #Parts >= 2 then
                local Value = Parts[2]:sub(2)
                if Value:sub(1, 3) == "rgb" then
                    local ColorParts = string.split(Value:sub(5, -2), ",")
                    Value = ColorParts
                elseif Value:sub(1, 3) == "key" then
                    local KeyParts = string.split(Value:sub(5, -2), ",")
                    if KeyParts[1] == "nil" and KeyParts[2] == "nil" then
                        KeyParts[1] = nil
                        KeyParts[2] = nil
                    end
                    Value = KeyParts
                elseif Value:sub(1, 4) == "bool" then
                    Value = Value:sub(6, -2) == "true"
                elseif Value:sub(1, 5) == "table" then
                    Value = string.split(Value:sub(7, -2), ",")
                elseif Value:sub(1, 6) == "string" then
                    Value = Value:sub(8, -2)
                elseif Value:sub(1, 6) == "number" then
                    Value = tonumber(Value:sub(8, -2))
                end
                Parsed[Parts[1]] = Value
            end
        end
        for Flag, Value in pairs(Parsed) do
            if self.Flags[Flag] then
                if typeof(self.Flags[Flag]) == "table" then
                    self.Flags[Flag]:Set(Value)
                else
                    self.Flags[Flag](Value)
                end
            end
        end
    end

    function NebulaX:SetOpen(State)
        if typeof(State) == "boolean" then
            self.IsOpen = State
            self.MainFrame.Visible = State
        end
    end

    function NebulaX:IsMouseOverFrame(Frame)
        local Pos, Size = Frame.AbsolutePosition, Frame.AbsoluteSize
        return Mouse.X >= Pos.X and Mouse.X <= Pos.X + Size.X and
               Mouse.Y >= Pos.Y and Mouse.Y <= Pos.Y + Size.Y
    end

    function NebulaX:ChangeAccent(Color)
        self.AccentColor = Color
        for _, Obj in pairs(self.ThemeObjects) do
            if Obj:IsA("Frame") or Obj:IsA("TextButton") then
                Obj.BackgroundColor3 = Color
            elseif Obj:IsA("TextLabel") or Obj:IsA("TextBox") then
                Obj.TextColor3 = Color
            elseif Obj:IsA("ImageLabel") or Obj:IsA("ImageButton") then
                Obj.ImageColor3 = Color
            elseif Obj:IsA("ScrollingFrame") then
                Obj.ScrollBarImageColor3 = Color
            end
        end
    end

    function NebulaX:Resize(Button, Frame)
        local Dragging, StartPos
        self:Connect(Button.MouseButton1Down, function(Input)
            Dragging = true
            StartPos = Input
        end)
        self:Connect(Mouse.Move, function(Input)
            if Dragging then
                local MousePos = UserInputService:GetMouseLocation()
                local X = math.clamp(MousePos.X - Frame.AbsolutePosition.X, 700, 9999)
                local Y = math.clamp((MousePos.Y - 36) - Frame.AbsolutePosition.Y, 750, 9999)
                Frame.Size = UDim2.new(0, X, 0, Y)
                for _, Tab in pairs(self.Tabs) do
                    Tab.Elements.Button.Size = UDim2.new(0, self.TabCount and ((((Frame.Size.X.Offset - 40) - ((self.TabCount - 1) * 5)) / self.TabCount)) - 5 or 80, 0, 24)
                end
            end
        end)
        self:Connect(UserInputService.InputEnded, function(Input)
            if Input.UserInputType == Enum.UserInputType.Touch then
                Dragging = false
            end
        end)
    end
end

-- Instance Creation with Theme Support
function NebulaX:NewInstance(Type, Theme)
    local Obj = Instance.new(Type)
    if Theme then
        table.insert(self.ThemeObjects, Obj)
        if Obj:IsA("Frame") or Obj:IsA("TextButton") then
            Obj.BackgroundColor3 = self.AccentColor
            if Obj:IsA("ScrollingFrame") then
                Obj.ScrollBarImageColor3 = self.AccentColor
            end
        elseif Obj:IsA("TextLabel") or Obj:IsA("TextBox") then
            Obj.TextColor3 = self.AccentColor
        elseif Obj:IsA("ImageLabel") or Obj:IsA("ImageButton") then
            Obj.ImageColor3 = self.AccentColor
        elseif Obj:IsA("UIStroke") then
            Obj.Color = self.AccentColor
        end
    end
    return Obj
end

-- Notification System
function NebulaX:Notification(Message, Duration, Color)
    local Notification = { Container = nil, Objects = {} }
    local Position = Vector2.new(20, 20)

    local NotifContainer = Instance.new("Frame", self.ScreenGui)
    NotifContainer.Name = "Notification"
    NotifContainer.Position = UDim2.new(0, Position.X, 0, Position.Y)
    NotifContainer.AutomaticSize = Enum.AutomaticSize.X
    NotifContainer.Size = UDim2.new(0, 0, 0, 18)
    NotifContainer.BackgroundTransparency = 1
    NotifContainer.ZIndex = 99999999
    Notification.Container = NotifContainer

    local Outline = Instance.new("Frame", NotifContainer)
    Outline.Name = "Outline"
    Outline.AutomaticSize = Enum.AutomaticSize.X
    Outline.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    Outline.BorderColor3 = Color3.fromRGB(0, 0, 0)
    Outline.Position = UDim2.new(0.01, 0, 0.02, 0)
    Outline.Size = UDim2.new(0, 0, 0, 18)
    Outline.BackgroundTransparency = 1
    table.insert(Notification.Objects, Outline)

    local Inline = Instance.new("Frame", Outline)
    Inline.Name = "Inline"
    Inline.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
    Inline.BorderSizePixel = 0
    Inline.Position = UDim2.new(0, 1, 0, 1)
    Inline.Size = UDim2.new(1, -2, 1, -2)
    Inline.BackgroundTransparency = 1
    table.insert(Notification.Objects, Inline)

    local Value = Instance.new("TextLabel", Inline)
    Value.Name = "Value"
    Value.FontFace = self.UIFont
    Value.RichText = true
    Value.Text = Message
    Value.TextColor3 = Color3.fromRGB(255, 255, 255)
    Value.TextSize = 12
    Value.TextStrokeTransparency = 0
    Value.TextXAlignment = Enum.TextXAlignment.Left
    Value.AutomaticSize = Enum.AutomaticSize.X
    Value.BackgroundTransparency = 1
    Value.Size = UDim2.new(0, 0, 1, 0)
    Value.TextTransparency = 1
    table.insert(Notification.Objects, Value)

    local UIPadding = Instance.new("UIPadding", Value)
    UIPadding.PaddingLeft = UDim.new(0, 6)
    UIPadding.PaddingRight = UDim.new(0, 6)
    UIPadding.PaddingTop = UDim.new(0, 2)

    local Accent = Instance.new("Frame", Outline)
    Accent.Name = "Accent"
    Accent.BackgroundColor3 = Color or self.AccentColor
    Accent.BorderSizePixel = 0
    Accent.Size = UDim2.new(1, 0, 0, 2)
    Accent.BackgroundTransparency = 1
    table.insert(Notification.Objects, Accent)

    function Notification:Remove()
        table.remove(self.Notifications, table.find(self.Notifications, Notification))
        self:UpdateNotificationPositions()
        task.wait(0.5)
        Notification.Container:Destroy()
    end

    task.spawn(function()
        Outline.AnchorPoint = Vector2.new(1, 0)
        for _, Obj in pairs(Notification.Objects) do
            if Obj:IsA("Frame") then
                TweenService:Create(Obj, TweenInfo.new(1, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {BackgroundTransparency = 0}):Play()
            elseif Obj:IsA("UIStroke") then
                TweenService:Create(Obj, TweenInfo.new(1, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Transparency = 0.8}):Play()
            end
        end
        TweenService:Create(Outline, TweenInfo.new(1, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {AnchorPoint = Vector2.new(0, 0)}):Play()
        TweenService:Create(Value, TweenInfo.new(1, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {TextTransparency = 0}):Play()
        task.wait(Duration)
        TweenService:Create(Outline, TweenInfo.new(1, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {AnchorPoint = Vector2.new(1, 0)}):Play()
        for _, Obj in pairs(Notification.Objects) do
            if Obj:IsA("Frame") then
                TweenService:Create(Obj, TweenInfo.new(1, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {BackgroundTransparency = 1}):Play()
            elseif Obj:IsA("UIStroke") then
                TweenService:Create(Obj, TweenInfo.new(1, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Transparency = 1}):Play()
            end
        end
        TweenService:Create(Value, TweenInfo.new(1, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {TextTransparency = 1}):Play()
    end)

    task.delay(Duration, function()
        Notification:Remove()
    end)

    table.insert(self.Notifications, Notification)
    NotifContainer.Position = UDim2.new(0, Position.X, 0, Position.Y + (#self.Notifications * 25))
    self:UpdateNotificationPositions()

    return Notification
end

function NebulaX:UpdateNotificationPositions()
    for i, Notif in pairs(self.Notifications) do
        TweenService:Create(Notif.Container, TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Position = UDim2.new(0, 20, 0, 20 + (i * 25))
        }):Play()
    end
end

            Value = math.clamp(NebulaX:Round(Value, Slider.Decimals), Slider.Min, Slider.Max)
            Fill.Size = UDim2.new((Value - Slider.Min) / (Slider.Max - Slider.Min), 0, 1, 0)
            ValueLabel.Text = tostring(Value) .. Slider.Suffix

        return { Set = Update }

    function NebulaX.Sections:Keybind(Properties)
        local Keybind = {
            Name = Properties.Name or "Keybind",
            Flag = Properties.Flag or NebulaX:NextFlag(),
            Default = Properties.Default or Enum.KeyCode.Unknown,
            UseKey = Properties.UseKey or false,
            Callback = Properties.Callback or function() end
        }

        local Frame = Instance.new("Frame", self.Elements.Frame)
        Frame.Name = "Keybind"
        Frame.BackgroundTransparency = 1
        Frame.Size = UDim2.new(1, 0, 0, 24)

        local Label = Instance.new("TextLabel", Frame)
        Label.Name = "Label"
        Label.FontFace = NebulaX.UIFont
        Label.Text = Keybind.Name
        Label.TextColor3 = Color3.fromRGB(255, 255, 255)
        Label.TextSize = 12
        Label.TextXAlignment = Enum.TextXAlignment.Left
        Label.BackgroundTransparency = 1
        Label.Size = UDim2.new(1, -60, 0, 24)

        local KeyButton = Instance.new("TextButton", Frame)
        KeyButton.Name = "KeyButton"
        KeyButton.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
        KeyButton.BorderColor3 = Color3.fromRGB(0, 0, 0)
        KeyButton.Position = UDim2.new(1, -60, 0, 0)
        KeyButton.Size = UDim2.new(0, 60, 0, 24)
        KeyButton.Text = Keybind.Default.Name or "None"
        KeyButton.FontFace = NebulaX.UIFont
        KeyButton.TextColor3 = Color3.fromRGB(255, 255, 255)
        KeyButton.TextSize = 12

        local Binding = false
        NebulaX:Connect(KeyButton.MouseButton1Click, function()
            Binding = true
            KeyButton.Text = "..."
        end)

        NebulaX:Connect(UserInputService.InputBegan, function(Input)
            if Binding then
                local Key = Keybind.UseKey and Input.KeyCode or Input.UserInputType
                if Key ~= Enum.KeyCode.Escape then
                    KeyButton.Text = NebulaX.Keybinds[Key] or Key.Name
                    NebulaX.Flags[Keybind.Flag] = Key
                    Keybind.Callback(Key)
                else
                    KeyButton.Text = "None"
                    NebulaX.Flags[Keybind.Flag] = nil
                    Keybind.Callback(nil)
                end
                Binding = false
            end
        end)

        if Keybind.Default ~= Enum.KeyCode.Unknown then
            KeyButton.Text = NebulaX.Keybinds[Keybind.Default] or Keybind.Default.Name
            NebulaX.Flags[Keybind.Flag] = Keybind.Default
            Keybind.Callback(Keybind.Default)
        end

        return { Set = function(Key) 
            KeyButton.Text = NebulaX.Keybinds[Key] or Key.Name
            NebulaX.Flags[Keybind.Flag] = Key
            Keybind.Callback(Key)
        end }
    end
end

local CurrentConfigs = {}

local function UpdateConfigList()
    local List = {}
    for _, File in ipairs(listfiles("NebulaX/Configs")) do
        local FileName = File:match("NebulaX/Configs/(.+)%.cfg")
        if FileName then
            table.insert(List, FileName)
        end
    end
    if #List ~= #CurrentConfigs or table.concat(List) ~= table.concat(CurrentConfigs) then
        CurrentConfigs = List
        ConfigList:Refresh(List)
    end
end

ConfigSection:Button({ Name = "Create", Callback = function()
    local ConfigName = Library.Flags.SettingsConfigurationName
    if ConfigName and ConfigName ~= "" and not isfile("NebulaX/Configs/" .. ConfigName .. ".cfg") then

        UpdateConfigList()
    end
end})
